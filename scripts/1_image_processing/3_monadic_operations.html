Monadic operations#

# Imports
import cv2
import numpy as np
import matplotlib.pyplot as plt

# Load image
image = cv2.imread('../../data/lenna_test_image.png')

# Convert BGR-space to RGB-space (OpenCV loads images in BGR-space)
image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)

# Print the shape, range, and data type of the image
print("The image is a matrix with dimensions: " + str(np.shape(image)) + ' which means 512 by 512 pixels and three channels')
print("The image values range from " + str(np.min(image)) + ' (minimum pixel value) to ' + str(np.max(image)) + ' (maximum pixel value)')
print("The datatype of the pixelvalues is stored as: " + str(type(image[0,0,0])) + ' (integers)')

# Show image
plt.imshow(image)
plt.show()

The image is a matrix with dimensions: (512, 512, 3) which means 512 by 512 pixels and three channels
The image values range from 3 (minimum pixel value) to 255 (maximum pixel value)
The datatype of the pixelvalues is stored as: <class 'numpy.uint8'> (integers)

Changing the pixel data type#

# Changing pixel type (convert to floats and normalize image)
image_double = cv2.normalize(image.astype('float'), None, 0.0, 1.0, cv2.NORM_MINMAX)

# Print the shape, range, and data type of the image
print("The image is a matrix with dimensions: " + str(np.shape(image_double)) + ' which means 512 by 512 pixels and three channels')
print("The image values range from " + str(np.min(image_double)) + ' (minimum pixel value) to ' + str(np.max(image_double)) + ' (maximum pixel value)')
print("The data type of the pixel values is stored as: " + str(type(image_double[0,0,0])) + ' (floats)')

# Show image
plt.imshow(image_double)
plt.show()

The image is a matrix with dimensions: (512, 512, 3) which means 512 by 512 pixels and three channels
The image values range from 0.0 (minimum pixel value) to 1.0 (maximum pixel value)
The data type of the pixel values is stored as: <class 'numpy.float64'> (floats)

Convert to grayscale#

# Converting to grayscale can be done using a linear combination of the three channels
gray = 0.2989 * image[:,:,2] + 0.5870 * image[:,:,1] + 0.1140 * image[:,:,0]

# Show image
plt.imshow(gray, cmap="gray")
plt.show()

Adjusting the contrast#

# Adjusting contrast by multiplying each pixel by two
image_contr = 2*image_double

# Show both images
plt.subplot(1, 2, 1)
plt.imshow(image_double)
plt.subplot(1, 2, 2)
plt.imshow(image_contr)
plt.show()

Clipping input data to the valid range for imshow with RGB data ([0..1] for floats or [0..255] for integers).

Adjusting the brightness#

# Adjusting the brightness by adding 0.2 to each pixel
image_bright = image_double + 0.2

# Show both images
plt.subplot(1, 2, 1)
plt.imshow(image_double)
plt.subplot(1, 2, 2)
plt.imshow(image_bright)
plt.show()

Clipping input data to the valid range for imshow with RGB data ([0..1] for floats or [0..255] for integers).
