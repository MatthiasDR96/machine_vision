BGR-space#

# Load image in BGR-space (OpenCV loads images in BGR-space by default)
image = cv2.imread("../../data/colorspaces.png")
print(np.shape(image))
plt.imshow(image)
plt.show()

RGB-space#

# Convert to RGB
rgb = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
print(np.shape(rgb))
plt.imshow(rgb)
plt.show()

Gray-space#

# Convert to grayscale
gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
print(np.shape(gray))
plt.imshow(gray, cmap='gray')
plt.show()

HSV-space#

# Convert to HSV
hsv = cv2.cvtColor(image, cv2.COLOR_BGR2HSV)
print(np.shape(hsv))
plt.imshow(hsv)
plt.show()

Masking#

# Masking
lower_green = np.array([100, 0, 0]) # Lower RGB-threshold
upper_green = np.array([120, 255, 255]) # Upper RGB-threshold
mask = cv2.inRange(hsv, lower_green, upper_green)
print(np.shape(mask))
plt.imshow(mask, cmap='gray')
plt.show()
